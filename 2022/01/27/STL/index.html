unique

将序列中不重复的相邻元素（即与前一个数不同）放在前面，需要先排序

int arr[] = {3,2,2,1,4}, n = sizeof arr/ sizeof 1;
for (int a : arr) cout<<a<<' ';
// 3 2 2 1 4 
sort(arr, arr+n);
// 1 2 2 3 4
n = unique(arr, arr+n) - arr;
for (int a : arr) cout<<a<<' ';
// 1 2 3 4 4
for (int i = 0; i < n; i++) cout<<arr[i]<<' ';
// 1 2 3 4
